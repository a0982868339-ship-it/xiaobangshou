## 💻 技术实现建议

### 1. 退出登录实现

```javascript
// mobile-user/src/views/Settings.vue
const handleLogout = () => {
  showConfirmDialog({
    title: '提示',
    message: '确定要退出登录吗？'
  }).then(() => {
    // 清除token
    localStorage.removeItem('token')
    // 清除用户信息
    localStorage.removeItem('userInfo')
    // 跳转登录页
    router.replace('/login')
    showToast('已退出登录')
  })
}
```

---

### 2. 修改密码实现

**后端接口**：
```javascript
// POST /api/users/change-password
exports.changePassword = async (req, res) => {
  const { oldPassword, newPassword } = req.body;
  const userId = req.user.id;
  
  // 1. 验证旧密码
  const [users] = await db.query('SELECT password FROM users WHERE id = ?', [userId]);
  if (!users[0].password || !bcrypt.compareSync(oldPassword, users[0].password)) {
    return sendError(res, '旧密码错误');
  }
  
  // 2. 加密新密码
  const hashedPassword = bcrypt.hashSync(newPassword, 10);
  
  // 3. 更新密码
  await db.query('UPDATE users SET password = ? WHERE id = ?', [hashedPassword, userId]);
  
  sendSuccess(res, null, '密码修改成功');
};
```

---

### 3. 服务者接单开关

**后端接口**：
```javascript
// PUT /api/providers/toggle-status
exports.toggleStatus = async (req, res) => {
  const userId = req.user.id;
  const { status } = req.body; // 0-休息中, 1-接单中
  
  await db.query(
    'UPDATE providers SET status = ? WHERE user_id = ?',
    [status, userId]
  );
  
  sendSuccess(res, null, status === 1 ? '已开启接单' : '已关闭接单');
};
```

## 📊 数据库表设计

### 用户设置表

```sql
CREATE TABLE `user_settings` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `notification_order` TINYINT DEFAULT 1 COMMENT '订单通知',
  `notification_activity` TINYINT DEFAULT 1 COMMENT '活动通知',
  `notification_system` TINYINT DEFAULT 1 COMMENT '系统通知',
  `notification_sound` TINYINT DEFAULT 1 COMMENT '声音提醒',
  `notification_vibrate` TINYINT DEFAULT 1 COMMENT '震动提醒',
  `dnd_enabled` TINYINT DEFAULT 0 COMMENT '勿扰模式',
  `dnd_start_time` TIME DEFAULT '22:00:00' COMMENT '勿扰开始时间',
  `dnd_end_time` TIME DEFAULT '08:00:00' COMMENT '勿扰结束时间',
  `privacy_show_phone` TINYINT DEFAULT 0 COMMENT '是否公开手机号',
  `dark_mode` TINYINT DEFAULT 0 COMMENT '深色模式: 0-关闭, 1-开启, 2-跟随系统',
  `language` VARCHAR(10) DEFAULT 'zh-CN' COMMENT '语言',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户设置表';
```
